  Race_ID SERIAL PRIMARY KEY,
  Name TEXT
  );
 CREATE TABLE First_names (
   First_name_ID SERIAL PRIMARY KEY,
   Gender TEXT,
    Race_ID INT REFERENCES Races(Race_ID),
    Name TEXT
    );
  CREATE TABLE Second_names (
    Second_name_ID SERIAL PRIMARY KEY,
    Race_ID int REFERENCES Races(Race_ID),
    Name TEXT
    );
